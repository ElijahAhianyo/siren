```python
def siren_uniform_(tensor: torch.Tensor, mode: str='fan_in', c: float=6):
    fan = _calculate_correct_fan(tensor, mode)
    std = 1 / math.sqrt(fan)
    bound = math.sqrt(c) * std
    with torch.no_grad():
        return tensor.uniform_(-bound, bound)
```
